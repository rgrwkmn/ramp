 12 			An object containing all dependency loading functionality.

 16 			/**
 17 				Plugins should be defined with ADF.addPlugins.
 18 				An array of JS plugin files to be loaded. Each file path is relative to the JS directory 
 19 				containing this framework and the .js extension is not required.
 20 				@example ["plugins"] would load the script plugins.js if it exists in the same directory as
 21 					this script.
 22 				@private
 23 			*/
 24 			plugins: [],
 25 			/**
 26 				Modules should be defined with ADF.addModules. 
 27 				An array of JS modules to be loaded. Each file path is relative to the JS directory 
 28 				containing this framework and the .js extension is not required. It is recommended that
 29 				these files be in a subdirectory called modules. If the modules have their own JS plugin
 30 				dependencies, those should be required by the modules themselves, which is explained in the
 31 				module documentation. These modules are loaded synchronously, and various callbacks are 
 32 				available to each module so that they may run code immediately when they are ready, when the
 33 				DOM is ready, or when the framework is ready after all the modules have been loaded.
 34 				@private
 35 			*/
 36 			modules: [],
 37 			/**
 38 				Contains the current framework ready states.
 39 				@namespace
 40 			*/

 42 				/** True if the DOM ready event has fired */
 43 				domReady: false,
 44 				/** True if the framework has loaded all defined plugin files */
 45 				pluginsReady: false,
 46 				/** True if the framework has loaded all defined modules. */
 47 				modulesReady: false
 48 			},
 49 			/**
 50 				Called from ADF.deps.load, sets one load dependency status as true.
 51 				@private
 52 			*/
 53 			set: function(dep, initFunction) {
 54 				//console.log('ADF.deps.set '+dep);
 55 				if (this.status[dep] !== false) {
 56 					console.log(dep+' is not tracked or has alreay been loaded');
 57 				} else {
 58 					this.status[dep] = true;
 59 				}
 60 				this.ready(initFunction);
 61 			},
 62 			/**
 63 				Called from ADF.deps.set, tests if all values in ADF.deps.status are true and calls the 
 64 				provided initFunction if they are.
 65 				@private
 66 			*/
 67 			ready: function(initFunction) {
 68 				var s = this.status;
 69 				for (var i in s) {
 70 					if (s.hasOwnProperty(i)) {
 71 						if (s[i] !== true) { return false; } 
 72 					}
 73 				}
 74 				initFunction();
 75 			},
 76 			/**
 77 				Starts the framework load sequence:
 78 				<ol>
 79 					<li>plugins</li>
 80 					<li>modules</li>
 81 				</ol>
 82 				The DOM ready event is triggered by jQuery asynchronously while the plugins and modules are
 83 				loaded synchronously by require.js. After modules are loaded by require, each module is 
 84 				assimilated by the framework using ADF.addModule.
 85 				@private
 86 			*/
 87 			load: function(initFunction) {
 88 				require(ADF.deps.plugins, function() {
 89 					ADF.deps.set('pluginsReady', initFunction);
 90 					
 91 					require(ADF.deps.modules, function() {
 92 						var mods = arguments;
 93 						var l = mods.length;
 94 						for (var i = 0; i < l; i++) {
 95 							ADF.addModule.call(ADF, mods[i]);
 96 						}
 97 						ADF.deps.set('modulesReady', initFunction);
 98 					});
 99 				});
100 
101 				$(function() {
102 					ADF.deps.set('domReady', initFunction);
103 					ADF.callModulesdomReady(ADF);
104 				});			
105 			}

133 			An object that stores all the loaded modules.
134 			@private

136 		modules: {},
137 		/**
138 			Called in a loop after the modules defined in ADF.deps.modules have been loaded with 
139 			require.js. Adds a module to the framework, storing it in ADF.modules using the module's name.
140 			If no name is defined addModule throws a warning to the console and returns. If the module
141 			has a name then addModule runs some business logic (ADF.getTileCallbacks) and fires the 
142 			appropriate ready callbacks if the module has them starting first with moduleReady and then
143 			domReady if the DOM is ready.
144 			@private
145 			@param {Module}
146 		*/
147 		addModule: function(m) {
148 			if (!m) { console.warn('ADF.addModule: no module? '+m); return; }
149 			if (!m.name) { console.warn('ADF.addModule: module must have a name'); console.log(m); return; }
150 			//console.log('ADF.addModule '+m.name);
151 			//console.log(m);
152 			this.modules[m.name] = m;
153 			m.domReadyCalled = false;
154 			
155 			//business logic module assimilation code
156 			//this.getTileCallbacks(m);
157 
158 			if (typeof m.moduleReady === 'function') {
159 				m.moduleReady.call(m, this);
160 			}
161 			if (this.deps.status.domReady && typeof m.domReady === 'function') {
162 				m.domReady.call(m, this);
163 				m.domReadyCalled = true;
164 			}
165 
166 			return this;
167 		},
168 		/**
169 			Calls all module domReady events if they haven't been called already.
170 			@private
171 		*/

173 			this.eachModule(function(m) {
174 				if (typeof m.domReady === 'function' && !m.domReadyCalled) {
175 					m.domReady(ADF);
176 				}
177 			});
178 
179 			return this;
180 		},
181 		/**
182 			Loops through the modules and executes a callback function for each one.
183 			@private
184 		*/

186 			console.log('eachModule');
187 			if (typeof f !== 'function') { console.error('ADF.eachModule: no callback function'); return; }
188 			var mods = this.modules;
189 			for (var i in mods) {
190 				if (mods.hasOwnProperty(i)) {
191 					console.log('call '+mods[i].name+' module ready');

193 				}
194 			}

199 			Called after all dependencies have been loaded (plugins, modules and DOM ready). Loops through

201 			@private
202 		*/

204 			console.log('ADF.init');
205 			var Framework = this;
206 			this.eachModule(function(m) {
207 				if (typeof m.ready === 'function') { m.ready(Framework); }
208 			});
209 
210 			return this;

212 		/**
213 			Calls a loaded module's API method with provided options. Use this in modules when you need
214 			to call API methods on other modules. It is recommended to also use this from a module to call
215 			its own API methods, since modules could eventually be unloaded or modified in a way that only
216 			the framework can track. API methods are called with the module's API object as the scope, so
217 			using the keyword "this" in API methods refers to the API object. If call returns null, the 
218 			API call failed, otherwise it will return whatever the API method returns, so it is 
219 			recommended to NOT return null from your API methods since a module that checks for null will
220 			assume that the call failed. Module API methods cannot be called until the module is loaded,
221 			so only make module API calls after the framework ready event has been called.
222 			@public
223 			@param {String} api A string defining the the module and API method separated by a period like so:
224 				"moduleName.apiMethod". See the Module documentation for more info on defining and calling
225 				module APIs.
226 			@param {Anything} [options] Optional argument to pass to the API method. Could be anything but
227 				it is recommended that it be an object for extensibility.
228 			@example Framework.call('coolModule.awesomeMethod', { radness: "extreme" });
229 		*/
230 		call: function(api, options) {
231 			var api = this.validateModFuncStr(api, 'ADF.call: first arg must be a string formatted thusly: "moduleName.methodName". You provided '+api);
232 			if (!api) { return null; } 
233 
234 			var moduleName = api[0];
235 			var module = this.modules[moduleName];
236 			if (typeof module === 'undefined') { console.warn('ADF.call: module '+moduleName+' does not exist.'); return null; }
237 
238 			var methodName = api[1];
239 			var method = module.api[methodName];
240 			if (typeof method !== 'function') { console.warn('ADF.call: module '+moduleName+' method '+methodName+' does not exist.'); return null; }
241 
242 			return method.call(module.api, this, options);
243 		},
244 		/**
245 			Define a callback function to be executed when a module publishes an event. The callback 
246 			function is called in the global scope.
247 			@see ADF.publish
248 			@param {String} event A string representing the module and event to listen for in the format
249 			"moduleName.eventName".
250 		*/
251 		subscribe: function(event, callback) {
252 			event = this.validateModFuncStr(event, 'ADF.subscribe: first arg must be a string formatted thusly: "moduleName.eventName". You provided '+event);
253 			if (!event) { return; }
254 
255 			if (typeof callback !== 'function') { console.warn('ADF.subscribe: a callback function is required'); return; }
256 
257 			var module = event[0];
258 			event = event[1];
259 			if (typeof this.subscriptions[module] === 'undefined') {
260 				this.subscriptions[module] = {};
261 			}
262 			if (typeof this.subscriptions[module][event] === 'undefined') {
263 				this.subscriptions[module][event] = [callback];
264 			} else {
265 				this.subscriptions[module][event].push(callback);
266 			}
267 
268 			return this;
269 		},
270 		/**
271 			Publish a module event. If any other modules have subscribed to the event, their callback
272 			functions will be executed with the additional arguments that you supply. Note that due to 
273 			the format of the first argument ("moduleName.eventName") it is technically possible to 
274 			publish	any module's events from any other--which is not recommended.
275 			@see ADF.subscribe
276 			@param {String} event A string representing the module and event being published in the format
277 			"moduleName.eventName".
278 			@param {Any} anything Each subsequent argument given to publish will be passed along to 
279 			functions that are subscribed to this event.
280 			@example 
281 			//from some module that wants to hook into an event
282 			Framework.subscribe('coolModule.awesomeEvent', function(one, two three) {
283 				//do stuff
284 			});
285 			//from coolModule calling the event
286 			Framework.publish('coolModule.awesomeEvent', one, two, three);
287 		*/
288 		publish: function() {
289 			var args = Array.prototype.slice.call(arguments);
290 			var event = args.shift();
291 			event = this.validateModFuncStr(event, 'ADF.publish: first arg must be a string formatted thusly: "moduleName.eventName". You provided '+event);
292 			var module = event[0];
293 			event = event[1];
294 			if (typeof this.subscriptions[module] === 'undefined') { return; }
295 			if (this.subscriptions[module][event] instanceof Array) {
296 				var events = this.subscriptions[module][event];
297 				var l = events.length;
298 				for (var i = 0; i < l; i++) {
299 					events[i].apply(null, args);
300 				}
301 			}
302 			
303 			return this;
304 		},
305 		/**
306 			Validates the module.function strings used by ADF.call, ADF.publish and ADF.subscribe.
307 			@param {String} s The module.function string to validate.
308 			@param {String} e The error message to console warn if s is not valid.
309 			@returns false if invalid, ["moduleName", "functionName"] if valid.
310 		*/
311 		validateModFuncStr: function(s, e) {
312 			if (typeof s !== 'string') { console.warn(e); return false; }
313 			if (!s.match(/.*\..*/)) { console.warn(e); return false; }
314 
315 			s = s.split('.');
316 			if (s.length !== 2) { console.warn(e); return false; }
317 
318 			return s;
319 		}
